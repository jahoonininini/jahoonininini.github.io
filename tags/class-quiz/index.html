Code

class House:
    # 매물을 초기화
    def __init__(self, location, house_type, deal_type, price, complection_year):
        self.location = location
        self.house_type = house_type
        self.deal_type = deal_type
        self.price = price
        self.complection_year = complection_year
        
        
        
    # 매물 정보 표시
    def show_detail(self):
        print(self.location, self.house_type, self.deal_type, self.price, self.complection_year)
        
        
a = House("동구", "아파트", "매매", "5억", "2020년")
a.show_detail()

b = House("달성군", "오피스텔", "전세", "3억", "2021년")
b.show_detail()

c = House("북구", "빌라", "월세", "500/30", "2019")
c.show_detail()

houses = []
house1 = House("동구", "아파트", "매매", "5억", "2020년")
house2 = House("달성군", "오피스텔", "전세", "3억", "2021년")
house3 = House("북구", "빌라", "월세", "500/30", "2019")

houses.append(house1)
houses.append(house2)
houses.append(house3)

print("총 {0} 곳의 매물이 있습니다.".format(len(houses)))

for house in houses:
    house.show_detail()
